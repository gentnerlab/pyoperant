% --- Glyph fallbacks --------------------------------------------------------
% The default Latin Modern text font lacks these glyphs; map them onto math
% equivalents that render under XeTeX/Tectonic. Each is wrapped in
% \texorpdfstring so it is also safe inside section headings (hyperref uses
% the ASCII form for PDF bookmarks). Prime characters (′ ″) are deliberately
% NOT mapped: unicode-math claims them as active math primes, so they are
% written as plain ASCII quotes in the source instead.
\usepackage{amssymb}
\usepackage{newunicodechar}
\newunicodechar{→}{\texorpdfstring{\ensuremath{\rightarrow}}{->}}
\newunicodechar{Ω}{\texorpdfstring{\ensuremath{\Omega}}{ohm}}
\newunicodechar{μ}{\texorpdfstring{\ensuremath{\mu}}{u}}
\newunicodechar{µ}{\texorpdfstring{\ensuremath{\mu}}{u}}
\newunicodechar{−}{\texorpdfstring{\ensuremath{-}}{-}}
\newunicodechar{✔}{\texorpdfstring{\ensuremath{\checkmark}}{[y]}}
\newunicodechar{≈}{\texorpdfstring{\ensuremath{\approx}}{~}}

% --- Running headers / footers ---------------------------------------------
\usepackage{fancyhdr}
\usepackage{etoolbox}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}
\fancyhead[L]{\small\itshape PyOperant Lab Manual}
\fancyhead[R]{\small\itshape\nouppercase{\leftmark}}
\fancyfoot[C]{\thepage}
% opening pages and the TOC use the "plain" style — keep a centered page number
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}
% In this document the top body heading level (## chapters) maps to
% \subsection. Start each chapter on a fresh page, and carry the chapter
% title into the right-hand running head.
\let\origsubsection\subsection
\renewcommand{\subsection}{\clearpage\origsubsection}
\renewcommand{\subsectionmark}[1]{\markboth{#1}{}}

% --- Tables: clean academic, avoid margin overflow -------------------------
% pandoc renders pipe tables as longtable; render them a touch smaller with
% tighter column padding and slightly airier rows.
\AtBeginEnvironment{longtable}{\small\setlength{\tabcolsep}{4pt}}
\renewcommand{\arraystretch}{1.15}
% absorb minor over-wide text lines (long URLs / part numbers in prose)
\setlength{\emergencystretch}{3em}

% --- Code blocks (via listings): monospace, wrapping, light frame ----------
% pandoc's --listings routes fenced code through the listings package.
\usepackage{listings}
\lstset{
  basicstyle=\ttfamily\small,
  breaklines=true,               % wrap long command / URL lines
  breakatwhitespace=false,
  showstringspaces=false,
  frame=single,
  rulecolor=\color[gray]{0.85},
  framesep=5pt,
  xleftmargin=6pt, xrightmargin=6pt,
  backgroundcolor=\color[gray]{0.97},
  aboveskip=0.8em, belowskip=0.8em,
}
